def find_gtk2_rb
  if File.exist?( "#{Config::CONFIG["sitelibdir"]}/gtk2.rb" )
    puts "gtk2.rb.. yes"
  else
    if File.exist?( "#{Config::CONFIG["rubylibdir"]}/gtk2.rb" )
      puts "gtk2.rb.. yes"
    else
      return false
    end
  end
  return true
end
